engine/learningEngine.ts

Motor principal de aprendizado.

import { storeMemory } from "../storage/memoryStore"
import { detectPatterns } from "./patternDetector"

export async function learnFromInteraction(data: any) {

  const memory = {
    vetId: data.vetId,
    query: data.query,
    response: data.response,
    timestamp: new Date()
  }

  await storeMemory(memory)

  const patterns = detectPatterns(memory)

  return patterns

}
engine/memoryBuilder.ts

Constrói memória clínica resumida.

export function buildMemorySummary(records: any[]) {

  return records.map(r => {

    return {
      symptom: r.query,
      decision: r.response
    }

  })

}
engine/patternDetector.ts

Detecta padrões do veterinário.

export function detectPatterns(memory: any) {

  const patterns: any = {}

  if (memory.query.includes("vômito")) {

    patterns.commonCondition = "gastrointestinal"

  }

  return patterns

}

storage/memoryStore.ts

Armazena memória da IA.

const memoryDB: any[] = []

export async function storeMemory(memory: any) {

  memoryDB.push(memory)

}

export async function getMemories(vetId: string) {

  return memoryDB.filter(m => m.vetId === vetId)

}
integration/agentLearningHook.ts

Hook que conecta o aprendizado ao agente.

import { learnFromInteraction } from "../engine/learningEngine"

export async function agentLearningHook(data: any) {

  const patterns = await learnFromInteraction(data)

  console.log("Padrões detectados:", patterns)

}

Use esse hook sempre após resposta da IA.

Exemplo de integração no agente
import { agentLearningHook } from "@/ai-learning/integration/agentLearningHook"

async function handleAIResponse(vetId, query, response) {

  await agentLearningHook({
    vetId,
    query,
    response
  })

}
